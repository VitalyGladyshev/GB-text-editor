30bool DocumentWindow::OpenFile(const QString &pathFileName)
31{
32 if (pathFileName.isEmpty())
33 {
34 emit SignalStatusBarMessage(tr("Path empty!"));
35 return false;
36 }
37
38 QFile file(pathFileName);
39
40 if(file.open(QIODevice::ReadOnly))
41 {
42 QGuiApplication::setOverrideCursor(Qt::WaitCursor);
43
44 QByteArray data = file.readAll();
45 file.close();
46 QMimeDatabase db;
47 const QString &mimeTypeName = db.mimeTypeForFileNameAndData(pathFileName, data).name();
48 if (mimeTypeName == u"text/html")
49 {
50
51 auto encoding = QStringDecoder::encodingForHtml(data);
52 QString str = QStringDecoder(encoding ? *encoding : QStringDecoder::Utf8)(data);
53 QUrl fileUrl =
54 pathFileName.startsWith(u':') ?
55 QUrl(pathFileName) : QUrl::fromLocalFile(pathFileName);
56 document()->setBaseUrl(fileUrl.adjusted(QUrl::RemoveFilename));
57 setHtml(str);
58
59 QFileInfo fi(pathFileName);
60 setSearchPaths({fi.path()});
61 setSource(fi.fileName());
62 }
63 else if (mimeTypeName == u"text/markdown")
64 setMarkdown(QString::fromUtf8(data));
65 else
66 setPlainText(QString::fromUtf8(data));
67
68 QFileInfo fi(pathFileName);
69 QString fileName = fi.fileName();
70 setWindowTitle(fileName);
71
72 QDir::setCurrent(fi.path());
73
74 _pathFileName = pathFileName;
75
76 emit SignalStatusBarMessage(
77 tr("File opened: \"%1\"").arg(QDir::toNativeSeparators(pathFileName)));
78 QGuiApplication::restoreOverrideCursor();
79 emit IsOpen(pathFileName);
80 return true;
81 }
82 else
83 {
84 emit SignalStatusBarMessage(
85 tr("File could not open: \"%1\"").arg(QDir::toNativeSeparators(pathFileName)));
86 return false;
87 }
88}

303// Поиск в тексте
304void DocumentWindow::Find(QString searchRequest, bool wholeText,
305 bool caseSensitive, bool backward)
306{
307 bool found = false;
308
309 QTextDocument::FindFlags flags;
310 if(wholeText)
311 flags |= QTextDocument::FindWholeWords;
312 if(caseSensitive)
313 flags |= QTextDocument::FindCaseSensitively;
314 if(backward)
315 flags |= QTextDocument::FindBackward;
316
317 QTextCursor cursor = textCursor();
318 cursor.clearSelection();
319
320 cursor = document()->find(searchRequest, cursor, flags);
321 if(!cursor.isNull())
322 {
323 found = true;
324 if (backward)
325 {
326 QTextCursor saveCursor = cursor;
327 cursor.setPosition(cursor.position(), QTextCursor::MoveAnchor);
328 cursor.setPosition(saveCursor.position()-searchRequest.length(),
329 QTextCursor::KeepAnchor);
330 setTextCursor(cursor);
331 }
332 else
333 setTextCursor(cursor);
334
335 activateWindow();
336 }
337
338 if (!found)
339 QMessageBox::information(this, tr("Find"), tr("Sequence not found!"));
340}

417 image->load(pathFileName);
418 relativePath = pathFileName;

420
421 QTextDocument* textDocument = document();

423 relativePath, QVariant(*image));
424 QTextCursor cursor = textCursor();
425 QTextImageFormat imageFormat;
426 imageFormat.setWidth(image->width());
427 imageFormat.setHeight(image->height());

451//метод изменяет отсутп в соответствии с заданным значением
452//(положительное число -увеличение, отрицательно число - уменьшение)
453void DocumentWindow::ModifyIndentation(int amount)
454{
455 QTextCursor cursor = this->textCursor();
456 cursor.beginEditBlock();
457 if (cursor.currentList())
458 {
459 QTextListFormat listFmt = cursor.currentList()->format();
460 QTextCursor above(cursor);
461 above.movePosition(QTextCursor::Up);
462 if (above.currentList() &&
463 listFmt.indent() + amount == above.currentList()->format().indent())
464 above.currentList()->add(cursor.block());
465 else
466 {
467 listFmt.setIndent(listFmt.indent() + amount);
468 cursor.createList(listFmt);
469 }
470 }
471 else
472 {
473 QTextBlockFormat blockFmt = cursor.blockFormat();
474 blockFmt.setIndent(blockFmt.indent() + amount);
475 cursor.setBlockFormat(blockFmt);
476 }
477 cursor.endEditBlock();
478}

514// Устанавливает стиль текста
515void DocumentWindow::TextStyle(int styleIndex)
516{
517 QTextCursor cursor = this->textCursor();
518 QTextListFormat::Style style = QTextListFormat::ListStyleUndefined;
519 QTextBlockFormat::MarkerType marker = QTextBlockFormat::MarkerType::NoMarker;
520
521 switch (styleIndex) {
522 case 1:
523 style = QTextListFormat::ListDisc;
524 break;
525 case 2:
526 style = QTextListFormat::ListCircle;
527 break;
528 case 3:
529 style = QTextListFormat::ListSquare;
530 break;
531 case 4:
532 if (cursor.currentList())
533 style = cursor.currentList()->format().style();
534 else
535 style = QTextListFormat::ListDisc;
536 marker = QTextBlockFormat::MarkerType::Unchecked;
537 break;
538 case 5:
539 if (cursor.currentList())
540 style = cursor.currentList()->format().style();
541 else
542 style = QTextListFormat::ListDisc;
543 marker = QTextBlockFormat::MarkerType::Checked;
544 break;
545 case 6:
546 style = QTextListFormat::ListDecimal;
547 break;
548 case 7:
549 style = QTextListFormat::ListLowerAlpha;
550 break;
551 case 8:
552 style = QTextListFormat::ListUpperAlpha;
553 break;
554 case 9:
555 style = QTextListFormat::ListLowerRoman;
556 break;
557 case 10:
558 style = QTextListFormat::ListUpperRoman;
559 break;
560 default:
561 break;
562 }
563
564 cursor.beginEditBlock();
565
566 QTextBlockFormat blockFormat = cursor.blockFormat();
567
568 if (style == QTextListFormat::ListStyleUndefined)
569 {
570 blockFormat.setObjectIndex(-1);
571 int headingLevel = styleIndex >= 11 ? styleIndex - 11 + 1 : 0; // H1 to H6, or Standard
572 blockFormat.setHeadingLevel(headingLevel);
573 cursor.setBlockFormat(blockFormat);
574
575 int sizeAdjustment = headingLevel ? 4 - headingLevel : 0; // H1 to H6: +3 to -2
576 QTextCharFormat fmt;
577 fmt.setFontWeight(headingLevel ? QFont::Bold : QFont::Normal);
578 fmt.setProperty(QTextFormat::FontSizeAdjustment, sizeAdjustment);
579 cursor.select(QTextCursor::LineUnderCursor);
580 cursor.mergeCharFormat(fmt);
581 this->mergeCurrentCharFormat(fmt);
582 }
583 else
584 {
585 blockFormat.setMarker(marker);
586 cursor.setBlockFormat(blockFormat);
587 QTextListFormat listFmt;
588 if (cursor.currentList())
589 listFmt = cursor.currentList()->format();
590 else
591 {
592 listFmt.setIndent(blockFormat.indent() + 1);
593 blockFormat.setIndent(0);
594 cursor.setBlockFormat(blockFormat);
595 }
596 listFmt.setStyle(style);
597 cursor.createList(listFmt);
598 }
599
600 cursor.endEditBlock();
601}
