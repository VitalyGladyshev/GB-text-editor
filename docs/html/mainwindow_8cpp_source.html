202{

207 {
208
209 pDocument->show(); // showMaximized()

244// Метод создаёт экземпляр дочернего MDI окна документа
245DocumentWindow* MainWindow::CreateNewDocument()
246{
247 DocumentWindow* pDocument = new DocumentWindow(this);
248 _pMdiArea->addSubWindow(pDocument);
249 pDocument->setAttribute(Qt::WA_DeleteOnClose);
250 pDocument->setWindowTitle(tr("Unnamed Document") + "_" + QString::number(_iUnnamedIndex++));
251 pDocument->setWindowIcon(QPixmap(":/images/icons/filenew.png"));
252 connect(pDocument, SIGNAL(SignalStatusBarMessage(QString)),
253 this, SLOT(SlotStatusBarMessage(QString)));
254 connect(pDocument, &QTextEdit::copyAvailable, _pCutAct, &QAction::setEnabled);
255 connect(pDocument, &QTextEdit::copyAvailable, _pCopyAct, &QAction::setEnabled);
256 connect(pDocument, &DocumentWindow::IsOpen,this,&MainWindow::SlotOnOpen);
257 connect(pDocument, &DocumentWindow::IsClose,this,&MainWindow::SlotDelPath);
258 connect(pDocument, SIGNAL(backwardAvailable(bool)), _pBackwardAct, SLOT(setEnabled(bool)));
259 connect(pDocument, SIGNAL(forwardAvailable(bool)), _pForwardAct, SLOT(setEnabled(bool)));
260 return pDocument;

440 QFileDialog fileDialog(this, tr("Save as ODT"), QDir::currentPath());

443 fileDialog.setMimeTypeFilters(mimeTypes);
444 fileDialog.setDefaultSuffix("odt");

446 return;
447 const QString pathFileName = fileDialog.selectedFiles().constFirst();

450 return;

452 pDocument->SaveAsOdt(pathFileName);
453 }
